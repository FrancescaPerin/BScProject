\begin{prooftree}








\AxiomC{$p, s ;  \overset{p}{\vdash} p, t ; q, r$}
\RightLabel{$R\lor_{+}$}
\UnaryInfC{$p, s ;  \overset{p}{\vdash} (p \lor t) ; q, r$}


\AxiomC{$q ; p, s \overset{q}{\vdash} q, r ; $}

\AxiomC{$r ; p, s \overset{r}{\vdash} q, r ; $}
\RightLabel{$L\lor_{+}$}
\BinaryInfC{$(q \lor r) ; p, s \overset{(q \lor r)}{\vdash} q, r ; $}
\RightLabel{$L\rightarrow_{+}$}
\BinaryInfC{$((p \lor t) \rightarrow (q \lor r)) ; p, s \overset{(p \rightarrow (q \lor r))}{\vdash} q, r ; $}
\RightLabel{$L\neg_{-}$}
\UnaryInfC{$((p \lor t) \rightarrow (q \lor r)) ; p, \neg(r), s \overset{(p \rightarrow (q \lor r))}{\vdash} q ; $}
\RightLabel{$L\neg_{-}$}
\UnaryInfC{$((p \lor t) \rightarrow (q \lor r)) ; \neg(q), p, \neg(r), s \overset{(p \rightarrow (q \lor r))}{\vdash}  ; $}
\RightLabel{$L\land_{-}$}
\UnaryInfC{$((p \lor t) \rightarrow (q \lor r)) ; (\neg(r) \land s), \neg(q), p \overset{(p \rightarrow (q \lor r))}{\vdash}  ; $}
\RightLabel{$R\neg_{+}$}
\UnaryInfC{$((p \lor t) \rightarrow (q \lor r)) ; (\neg(r) \land s), \neg(q) \overset{(p \rightarrow (q \lor r))}{\vdash} \neg(p) ; $}
\RightLabel{$R\rightarrow_{+}$}
\UnaryInfC{$((p \lor t) \rightarrow (q \lor r)) ; (\neg(r) \land s) \overset{(p \rightarrow (q \lor r))}{\vdash} (\neg(q) \rightarrow \neg(p)) ; $}
\RightLabel{$R\rightarrow_{+}$}
\UnaryInfC{$((p \lor t) \rightarrow (q \lor r)) ;  \overset{(p \rightarrow (q \lor r))}{\vdash} ((\neg(r) \land s) \rightarrow (\neg(q) \rightarrow \neg(p))) ; $}
\end{prooftree}
